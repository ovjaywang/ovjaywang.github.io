        DateFormat df=new SimpleDateFormat("yyyy/MM/dd HH:mm:ss");

    }
string->date 
public static Date Trans(String str)
    {
        DateFormat df=new SimpleDateFormat("yyyy/MM/dd HH:mm:ss");
        Date d=null;
        try {
            d=df.parse(str);
        } catch (ParseException e) {
            e.printStackTrace();
        }
        if(d==null)
        {System.out.println("格式输入失败");}
        else
        {
            System.out.println("格式转换成功  "+d);
        }
        return d;
    }
